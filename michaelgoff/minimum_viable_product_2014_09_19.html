# Following are some helpful filtering and aggregation functions

# Restrict trip data to morning trips
def restrictHours(t, first_hour, last_hour):
    i = int(t[11:13]) # Extract the appropriate characters from the starttime field for the hour
    return i >= first_hour and i <= last_hour

# Apply an age bracket to the rider, given the birth year.  Assumes data from 2014
def age_bracket(age_str,brackets):
    try:
        a = int(age_str)
    except ValueError:
        return 'uxxxx'
    lower = '00'
    for i in range(len(brackets)):
        if a >= 2014 - brackets[i]:
            return 'u' + lower + str(brackets[i])
        lower = str(brackets[i]+1)
    if a >= 1901:
        return 'u' + lower + 'xx'
    return 'uxxxx'

############################################################

# Apply a sequence of filters to the data
def TripDataFilter(tripdata, filters):
    for f in filters:
        if f['type'] == 'RestrictHours':
            tripdata = tripdata[tripdata['starttime'].apply(lambda t: restrictHours(t,f['first'],f['last']))]
    return tripdata

def AddClassifiers(tripdata, classifiers):
    for c in classifiers:
        if c['t'] == "AgeBrackets": # Create age bracket classifiers where c['brackets'] contains the division points
            tripdata[c['colname']] = tripdata['birth year'].apply(lambda x: age_bracket(x,c['brackets']))
        if c['t'] == "Counter":
            tripdata[c['colname']] = 'COUNT'
    return tripdata   

# Create a dataframe that has counts of values of ref by index
def GroupByCount(data,index,column_names = None,refs=[]):
    gb = data.groupby([index]+refs)
    agg = gb[index].count()
    for i in range(len(refs)):
        agg = agg.unstack()
    agg = agg.fillna(0)
    if (column_names):
        agg.rename(columns=column_names,inplace=True)
    return agg

def GroupVariables(data,index,agg_pipeline):
    gb = data.groupby(index)
    ret = pd.DataFrame(gb[index].count())
    del ret[index]
    for a in agg_pipeline:
        col_names = None
        if 'column_names' in a:
            col_names = a['column_names']
        suffixes = ['','']
        if 'suffixes' in a:
            suffixes = a['suffixes']
        ret = ret.join(GroupByCount(data,index,refs=a['refs'],column_names = col_names),rsuffix = suffixes[1])
    return ret

# Read in the trip data and combine it into a dataframe with statistics
def read_data(y,m,filters=[],classifiers=[],groups=[],dfindex=''):
    y = str(y)
    m = str(m)
    if (len(m)==1):
        m = '0'+m
    cb2 = pd.read_csv("Data/"+y+"-"+m+" - Citi Bike trip data.csv",dtype=object)
    cb2 = TripDataFilter(cb2, filters)
    cb2 = AddClassifiers(cb2, classifiers)
    grouped = GroupVariables(cb2,dfindex,groups)
    return grouped

def ReadMultiData(dates,filters=[],classifiers=[],groups=[],dfindex=''):
    StationCount = read_data(dates[0][0],dates[0][1], filters=filters, classifiers=classifiers, groups=groups,dfindex=dfindex)
    for i in range(1,len(dates)):
        new_data = read_data(dates[i][0],dates[i][1], filters=filters, classifiers=classifiers,groups=groups,dfindex=dfindex)
        for col in new_data.columns.values:
            if col not in StationCount.columns.values:
                StationCount[col] = 0
        for col in StationCount.columns.values:
            if col not in new_data.columns.values:
                new_data[col] = 0
        StationCount += new_data
    return StationCount

# Extract and add the census block group and tract to each entry in the age table.  This will help with merging
def toBlockTract(geoid):
    geoid = str(geoid)
    try:
        tract = geoid[2:9]
    except ValueError:
        return "0"
    try:
        block = geoid[9:11] # Change to [9:12] to incorporate the block group
    except ValueError:
        return "0"
    return tract + '.' + block

# Build a linear model based selected inputs from a csv file
# The CSV file should provide data for all census blocks in the CitiBike network and
# be formatted in the matter of US Census files.
def BuildLinearModel(filename,agg_cols,name='',col_names=[]):
    raw_data = pd.read_csv(filename,dtype=object).ix[2:,:]
    raw_data["asdf"] = raw_data["GEO.id2"].apply(toBlockTract)
    # This operation is necessary to index on tracts rather than block groups
    for i in range(len(agg_cols)):
        for j in range(len(agg_cols[i])):
            raw_data[agg_cols[i][j]] = raw_data[agg_cols[i][j]].astype(float)
    sum_agg = []
    for i in range(len(agg_cols)):
        for j in range(len(agg_cols[i])):
            sum_agg.append(agg_cols[i][j])
    raw_data_tract = raw_data.ix[:,sum_agg+['asdf']].groupby('asdf').sum()
    StationCount_aug = CensusBlock.join(raw_data_tract)
    # Trim obviously bad values
    StationCount_aug = StationCount_aug[CensusBlock['COUNT']>0]
    global count
    count = count[CensusBlock['COUNT'] > 0]
    # Convert data frame columns from strings to floats
    float_conv = []
    for i in range(len(agg_cols)):
        for j in range(len(agg_cols[i])):
            float_conv.append(agg_cols[i][j])
    for col in float_conv:
        StationCount_aug[col] = StationCount_aug[col].astype(float)
    # Build the X dataframe for the linear regression
    new_df_dict = {}
    single_var_regressions = []
    for i in range(len(agg_cols)):
        new_col = StationCount_aug[agg_cols[i][0]]
        for j in range(1,len(agg_cols[i])):
            new_col = new_col + StationCount_aug[agg_cols[i][j]]
        new_df_dict['x'+str(i)] = new_col
        single_var_regressions.append(pd.DataFrame({'variable':new_col}))
    StationData = pd.DataFrame(new_df_dict)
    # Conduct the linear regression
    clf = linear_model.LinearRegression().fit(StationData,count)
    ret_value = [clf,[]]
    for i in range(len(single_var_regressions)):
        clf_single = linear_model.LinearRegression().fit(single_var_regressions[i],count)
        ret_value[1].append([clf_single,single_var_regressions[i]])
    return ret_value
