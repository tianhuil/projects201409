def noun_lis(sentence):
    Text=word_tokenize(sentence)
    Pos=pos_tag(Text)
    Nouns=[word for word,pos in Pos if pos=='NN']
    return Nouns
noun_lis('This movie has a good plot, but bad acting. Really bad acting.')

def noun_freq(sentence,freq):
    for noun in noun_lis(sentence):
        if noun in freq:
            freq[noun]+=1
        else: freq[noun]=1
    return freq
